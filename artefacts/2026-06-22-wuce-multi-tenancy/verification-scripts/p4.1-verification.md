## AC Verification

### AC1 — Prompt-cache keys distinct across tenants (same sessionId)
```
node -e "
var { buildCacheKey } = require('./src/web-ui/adapters/cache-key');
var keyA = buildCacheKey({ tenantId: 'org-a', sessionId: 'abc' });
var keyB = buildCacheKey({ tenantId: 'org-b', sessionId: 'abc' });
console.assert(keyA !== keyB, 'FAIL: keys must differ');
console.log('AC1:', keyA, '!==', keyB, '✓');
"
```

### AC2 — Cache key format `${tenantId}-${sessionId}`
```
node -e "
var { buildCacheKey } = require('./src/web-ui/adapters/cache-key');
var key = buildCacheKey({ tenantId: 'my-org', sessionId: 'xyz123' });
console.assert(key === 'my-org-xyz123', 'FAIL: expected my-org-xyz123, got ' + key);
console.log('AC2:', key, '✓');
"
```

### AC3 — Cache key falls back to sessionId when tenantId absent

### AC4 — Per-tenant rate-limit counters isolated
```
node -e "
var { createRateLimiter } = require('./src/web-ui/middleware/rate-limiter');
var rl = createRateLimiter({ maxRequests: 100, windowMs: 60000 });
var next = function() {};
var makeReq = function(tenantId) { return { session: { tenantId: tenantId, userId: '1', login: 'u' }, headers: {}, connection: { remoteAddress: '1.2.3.4' } }; };
for (var i = 0; i < 90; i++) rl(makeReq('org-a'), {}, next);
var res = { _code: null, writeHead: function(c) { this._code = c; }, end: function() {} };
rl(makeReq('org-b'), res, next);
console.assert(res._code !== 429, 'FAIL: org-b hit rate limit due to org-a');
console.log('AC4: org-b not rate-limited after org-a hammers ✓');
"
```

### AC5 — Rate-limit key includes tenantId

node -e "
