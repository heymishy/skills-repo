### AC1 — slugifyTenantId rejects path separators (forward and backward slash)
```
node -e "
var { slugifyTenantId } = require('./src/web-ui/adapters/repo-root');
var s1 = slugifyTenantId('../../etc/passwd');
var s2 = slugifyTenantId('org-a/../../../etc');
var s3 = slugifyTenantId('org-a\\\\..\\\\etc');
console.assert(!s1.includes('/') && !s1.includes('\\\\'), 'AC1a FAIL: ' + s1);
console.assert(!s2.includes('/') && !s2.includes('\\\\'), 'AC1b FAIL: ' + s2);
console.assert(!s3.includes('/') && !s3.includes('\\\\'), 'AC1c FAIL: ' + s3);
console.log('AC1:', s1, '/', s2, '/', s3, '✓');
"
```

### AC6 — Path-traversal guard fires on all adversarial slugs
```
node -e "
var path = require('path');
var os = require('os');
var { slugifyTenantId } = require('./src/web-ui/adapters/repo-root');
var base = os.tmpdir();
var corpus = ['../../etc/passwd','org-a/../../../etc','org-a\\\\..\\\\etc','%2e%2e%2f%2e%2e%2fetc%2fpasswd','%2e%2e%5c%2e%2e%5cetc','org-a\x00../../etc','org-a\x00','a'.repeat(300),'оrg-а','ｏrｇ-ａ',' ../../etc'];
corpus.forEach(function(input) {
  var slug = slugifyTenantId(input);
  var derived = path.resolve(path.join(base, slug));
  console.assert(derived.startsWith(path.resolve(base) + path.sep), 'AC6 FAIL for: ' + JSON.stringify(input) + ' -> slug=' + slug + ' -> ' + derived);
});
console.log('AC6: all ' + corpus.length + ' adversarial inputs safe ✓');
"
```
